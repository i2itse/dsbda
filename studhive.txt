C. Join Tables (student1 and student2)
SELECT s1.stu_id, s1.stu_name, s1.stu_branch, s2.subject, s2.marks
FROM student1 s1
JOIN student2 s2
ON s1.stu_id = s2.stu_id;




D. Students having marks greater than 80
SELECT s1.stu_id, s1.stu_name, s2.marks
FROM student1 s1
JOIN student2 s2
ON s1.stu_id = s2.stu_id
WHERE s2.marks > 80;
